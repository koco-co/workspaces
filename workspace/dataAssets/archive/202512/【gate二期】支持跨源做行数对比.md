```

2. 公共配置全局参数${year} = ’2025‘
3. 数据源下分别存在分区表如下：
doris:
CREATE TABLE IF NOT EXISTS example_db.composite_partitioned_table (
id            BIGINT,
name          VARCHAR(50),
age           INT,
salary        DOUBLE,
hire_date     DATE,
department_id INT
)
-- 只使用一级分区：按时间分区
PARTITION BY RANGE (hire_date) (
PARTITION p2024 VALUES LESS THAN (“2025-01-01“),
PARTITION p2025 VALUES LESS THAN (“2026-01-01“)
)
DISTRIBUTED BY HASH (id) BUCKETS 10
PROPERTIES (
“replication_num“ = “3“
);

INSERT INTO composite_partitioned_table VALUES
(1, '张三', 25, 15000.00, '2024-06-15', 101),  -- p2024
(2, '李四', 28, 18000.00, '2025-03-20', 102);  --p2025


hive:
CREATE TABLE employees_partitioned ( id INT, name STRING, age INT ) PARTITIONED BY (year STRING);

# 插入3条数据 INSERT INTO employees_partitioned PARTITION(year='2023') VALUES(1,'test1',11); INSERT INTO employees_partitioned PARTITION(year='2024') VALUES(2,'test2',12); INSERT INTO employees_partitioned PARTITION(year='2025') VALUES(3,'test3',13);
```
