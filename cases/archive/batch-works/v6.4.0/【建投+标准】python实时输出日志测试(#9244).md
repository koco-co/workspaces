from __future__ import print_function

if sys.version_info[0] < 3:
    # Py2：打印中文时编码为 UTF-8
    def uprint(s):
        sys.stdout.write((s if isinstance(s, str) else s.encode('utf-8')) + '\n')

    if num == 1:
        count += 1
        uprint(u"第{}步: 第1个轮盘从:{}柱->{}柱".format(count, a, c))

        hanoi_tower(num - 1, a, c, b)

        uprint(u"第{}步: 第{}个轮盘从:{}柱->{}柱".format(count, num, a, c))
        hanoi_tower(num - 1, b, a, c)
    time.sleep(1)


if __name__ == "__main__":
