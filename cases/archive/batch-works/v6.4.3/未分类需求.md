```
 DENSE_RANK()     OVER (ORDER BY total_amount DESC)        AS dense_rank_by_amount,
 FIRST_VALUE(total_amount) OVER (ORDER BY customer_id)     AS first_total_amount,
 LAST_VALUE(total_amount)  OVER (
 ORDER BY customer_id
 ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING
 ) AS last_total_amount,
 PERCENT_RANK()   OVER (ORDER BY total_amount)              AS pct_rank_amount,

 -- 日期函数
 current_date                                               AS today,
 current_timestamp                                          AS now_ts,
 date_trunc('day', first_txn_time)                          AS first_txn_day,
 EXTRACT(YEAR FROM first_txn_time)                           AS first_txn_year,
 FLOOR( EXTRACT(EPOCH FROM (now_ts - first_txn_time)) / 3600 ) AS hours_since_first_txn

FROM agg
ORDER BY total_amount DESC;
```
